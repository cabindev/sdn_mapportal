## โครงสร้างข้อมูล

```typescript
interface RegionData {
  district: string;      // ชื่อตำบล
  amphoe: string;        // ชื่ออำเภอ
  province: string;      // ชื่อจังหวัด
  zipcode: number;       // รหัสไปรษณีย์
  district_code: number; // รหัสตำบล
  amphoe_code: number;   // รหัสอำเภอ
  province_code: number; // รหัสจังหวัด
  zone: string;          // ภูมิภาค
}
```
